import javafx.application.Application; import javafx.geometry.Insets; import javafx.geometry.Pos; import javafx.scene.Scene; import javafx.scene.control.Button; import javafx.scene.control.Label; import javafx.scene.control.PasswordField; import javafx.scene.control.TextField; import javafx.scene.layout.VBox; import javafx.scene.paint.Color; import javafx.stage.Stage; public class LoginPage extends Application { @Override public void start(Stage primaryStage) { // Create the main container VBox loginContainer = new VBox(10); loginContainer.setPadding(new Insets(20)); loginContainer.setAlignment(Pos.CENTER); loginContainer.setStyle("-fx-background-color: #f4f4f4;"); // Create the title label Label titleLabel = new Label("Login"); titleLabel.setStyle("-fx-font-size: 20px; -fx-font-weight: bold;"); // Create email input field Label emailLabel = new Label("Email"); TextField emailField = new TextField(); emailField.setPromptText("Enter your email"); Label emailError = new Label(); emailError.setTextFill(Color.RED); emailError.setStyle("-fx-font-size: 12px;"); // Create password input field Label passwordLabel = new Label("Password"); PasswordField passwordField = new PasswordField(); passwordField.setPromptText("Enter your password"); Label passwordError = new Label(); passwordError.setTextFill(Color.RED); passwordError.setStyle("-fx-font-size: 12px;"); // Create login button Button loginButton = new Button("Login"); loginButton.setStyle("-fx-background-color: #007bff; -fx-text-fill: white; -fx-padding: 10px;"); loginButton.setOnAction(e -> validateLogin(emailField, passwordField, emailError, passwordError)); // Add components to the container loginContainer.getChildren().addAll(titleLabel, emailLabel, emailField, emailError, passwordLabel, passwordField, passwordError, loginButton); // Set up the scene and stage Scene scene = new Scene(loginContainer, 350, 400); primaryStage.setTitle("Login Page"); primaryStage.setScene(scene); primaryStage.show(); } private void validateLogin(TextField emailField, PasswordField passwordField, Label emailError, Label passwordError) { String email = emailField.getText().trim(); String password = passwordField.getText().trim(); emailError.setText(""); passwordError.setText(""); boolean isValid = true; if (email.isEmpty()) { emailError.setText("Email is required"); isValid = false; } else if (!email.contains("@")) { emailError.setText("Invalid email format"); isValid = false; } if (password.isEmpty()) { passwordError.setText("Password is required"); isValid = false; } else if (password.length() < 6) { passwordError.setText("Password must be at least 6 characters long"); isValid = false; } if (isValid) { System.out.println("Login successful!"); // You can replace this with further logic } } public static void main(String[] args) { launch(args); } }
